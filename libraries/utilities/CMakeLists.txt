list( APPEND CMAKE_MODULE_PATH "${CMAKE_SOURCE_DIR}/libraries/fc/GitVersionGen" )
include( GetGitRevisionDescription )
get_git_head_revision(GIT_REFSPEC STEEMIT_GIT_REVISION_SHA)
get_git_unix_timestamp(STEEMIT_GIT_REVISION_UNIX_TIMESTAMP)
git_describe(STEEMIT_GIT_REVISION_DESCRIPTION --tags)
if(NOT STEEMIT_GIT_REVISION_DESCRIPTION)
    set(STEEMIT_GIT_REVISION_DESCRIPTION "unknown")
endif(NOT STEEMIT_GIT_REVISION_DESCRIPTION)

file(GLOB HEADERS "include/steemit/utilities/*.hpp")

set(sources
   key_conversion.cpp
   string_escape.cpp
   tempdir.cpp
   words.cpp
   logging_config.cpp
   ${HEADERS})

configure_file("${CMAKE_CURRENT_SOURCE_DIR}/git_revision.cpp.in" "${CMAKE_CURRENT_BINARY_DIR}/git_revision.cpp" @ONLY)
list(APPEND sources "${CMAKE_CURRENT_BINARY_DIR}/git_revision.cpp")

add_library( steemit_utilities
             ${sources}
             ${HEADERS} )
target_link_libraries( steemit_utilities fc )
target_include_directories( steemit_utilities
                            PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/include" )
if (USE_PCH)
  set_target_properties(steemit_utilities PROPERTIES COTIRE_ADD_UNITY_BUILD FALSE)
  cotire(steemit_utilities)
endif(USE_PCH)

install( TARGETS
   steemit_utilities

   RUNTIME DESTINATION bin
   LIBRARY DESTINATION lib
   ARCHIVE DESTINATION lib
)
install( FILES ${HEADERS} DESTINATION "include/steemit/utilities" )
